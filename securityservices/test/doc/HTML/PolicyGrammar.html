policyFile

    :   ( declaration)*

        ( policy)*

    ;

 

declaration

    : setprefix

    | "Delete" TOKEN

    | "Agent" URI

    | "UserRole" TOKEN

    | "Servlet" TOKEN

    | "PlugInRole" TOKEN

    | "BlackBoardObject" TOKEN

    ;

 

setprefix

    : "PolicyPrefix" EQ URI  

    ;

 

policy

    : "Policy" TOKEN EQ LBRACK

         innerPolicy RBRACK

        conditionalAddendum

    ;

 

conditionalAddendum

    : "when" "operating" "mode" EQ TOKEN

    |

    ;

 

innerPolicy

    : "GenericTemplate"                genericPolicy

    | "AuditTemplate"                  audit

    | "BlackboardTemplate"             blackboardPolicy

    | "ServletAuthenticationTemplate"  servletAuthentication

    | "ServletUserAccessTemplate"      servletUserAccess

    ;

 

 

genericPolicy

    : "Priority" EQ INT COMMA

      URI "is" genericAuth

        "to" "perform" URI "as" "long" "as"

 

        genericTargets

       

    ;

 

genericAuth

    : "authorized"

    | "not" "authorized"

    ;

 

genericTargets

    : genericTarget genericMoreTargets

    ;

 

genericMoreTargets

    :

    | "and" genericTarget genericMoreTargets

    ;

 

genericTarget

 

    : "the" "value" "of" URI genericRestrictionType

        genericTargetModality

        genericRange

    ;

 

genericRestrictionType

    : "is" "a" "subset" "of" "the"

       

    | "contains" "at" "least" "one" "element" "from" "the"

       

    ;

 

genericTargetModality

    : "set"

    | "complement" "of" "the" "set"

    ;

 

genericRange

    : URI

    |

        LCURLY

        ( URI )*

        RCURLY

    ;

 

 

audit

   : "Require" "audit" "for" "all" "accesses" "to" "servlet"

        tokenList auditUsers

       

    ;

 

auditUsers

    : "by" "users" "in" "role" TOKEN

    |

    ;

 

 

/*

 * The Blackboard policy template: (e.g. A plugin in the role OpPlan can

 * add, remove, change, query objects of type OpPlan on the blackboard.

 */

 

blackboardPolicy

    : "A" "PlugIn" "in" "the" "role" TOKEN "can"

        tokenList "objects" "of" "type"

        tokenList

       

    ;

 

 

/*

 * The servlet authentication servlet (e.g. All users must use CertificateSSL

 * when accessing the servlet named PolicyServlet)

 */

 

servletAuthentication

    : "All" "users" "must" "use" tokenList "authentication" "when"

        "accessing" "the" "servlet" "named" TOKEN

       

    ;

 

 

 

/*

 * The Servlet Access template: (e.g. A user in role policyAdministrator is

 * allowed to access a servlet named PolicyServlet)

 */

servletUserAccess 

    :   "A" "user" "in" "role" TOKEN servletUserAccessModality

        "access" "a" "servlet" "named" TOKEN

       

    ;

 

servletUserAccessModality

    : "can"

    | "cannot"

   ;

 

 

 

/*

 * tokenList is used to represent a list of tokens.  It returns a set

 * consisting of the tokens.

 */

tokenList

    : TOKEN moreTokenList

    ;

 

moreTokenList

    : COMMA TOKEN  moreTokenList

    |

    ;
